{ STEP 1: Replace <FORMULA1> through <FORMULA4> with the formulas you wish to plot.  
  Note that Value1 through Value4 are numeric variables, temporary holding places for

inputs: 
	DefaultPrice(Close),
	{NormRatio(2),}
	FastLength(4),
	MidLength(14),
	SlowLength(30),
	TLLength(4), 
	ColorTLAngMaFa(Red), 
	ColorTLAngMaMd(Magenta), 
	ColorTLAngMaSl(Yellow), 
	ColorMaTLAngSl(Cyan), 
	ColorBase(Darkgray), 	
	Displace( 0 ) ;

variables:
	Price(DefaultPrice),
	FastAvgTLAngMa(0),
	MidAvgTLAngMa(0),
	SlowAvgTLAngMa(0),
	
	FastAvgMaTLAng(0),
	MidAvgMaTLAng(0),
	SlowAvgMaTLAng(0),
	double NormRatio(-1)  ;

{Value1 = LinearRegAngleFC( NormRatio*Price, TLLength ) ;}

once (Currentbar = 1) Begin NormRatio = GZF_GetSymbolPropDouble(symbol, "NORMRATIO") ; End ;
if LastBarOnChart then once print(symbol, ", NormRatio=", NormRatio);
	
switch(FastAvgTLAngMa[1])
	Begin 
		Case > 0 :
			Price = High ; Break ;
		Case < 0 :
			Price = Low ; Break ;
		Default : 
			Price = Close ;
	End ;

FastAvgTLAngMa = GZF_TLAngMA(Price, NormRatio, FastLength, TLLength); {AverageFC( Value1, FastLength ) ;}
MidAvgTLAngMa = GZF_TLAngMA(Price, NormRatio, MidLength, TLLength); 
SlowAvgTLAngMa = GZF_TLAngMA(Price, NormRatio, SlowLength, TLLength); {AverageFC( Value1, SlowLength ) ;}

MidAvgMaTLAng = GZF_MATLAng(Price, NormRatio, MidLength, TLLength);
SlowAvgMaTLAng = GZF_MATLAng(Price, NormRatio, SlowLength, TLLength);

Plot1( 90, "TLAngMAUp" ) ;
Plot2( 0, "TLAngMAFlat" ) ;
Plot3( -90, "TLAngMADn" ) ;
SetPlotColor( 1, ColorBase ) ;
SetPlotColor( 2, ColorBase ) ;
SetPlotColor( 3, ColorBase ) ;

Plot4( FastAvgTLAngMa, "TLAngMAFa" ) ;
SetPlotColor( 4, ColorTLAngMaFa ) ;

Plot5( MidAvgTLAngMa, "TLAngMAMd" ) ;
SetPlotColor( 5, ColorTLAngMaMd ) ;

Plot6( SlowAvgTLAngMa, "TLAngMASl" ) ;
SetPlotColor( 6, ColorTLAngMaSl ) ;

Plot7( SlowAvgMaTLAng, "MATLAngSl" ) ;
SetPlotColor( 7, ColorMaTLAngSl ) ;

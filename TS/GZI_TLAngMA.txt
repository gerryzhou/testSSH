inputs: 
	Price( Close ),
	NormRatio(2),
	FastLength( 4 ),
	SlowLength( 30 ),
	TLLength( 4 ), 
	ColorTLAngMaFa( Red ), 
	ColorTLAngMaSl( Blue ), 
	ColorMaTLAngSl( Cyan ), 
	ColorBase( Darkgreen ), 	
	Displace( 0 ) ;

variables:
	FastAvgTLAngMa( 0 ),
	SlowAvgTLAngMa( 0 ),
	SlowAvgMaTLAng( 0 )  ;

{Value1 = LinearRegAngleFC( NormRatio*Price, TLLength ) ;}

FastAvgTLAngMa = GZF_TLAngMA(Price, NormRatio, FastLength, TLLength); {AverageFC( Value1, FastLength ) ;}
SlowAvgTLAngMa = GZF_TLAngMA(Price, NormRatio, SlowLength, TLLength); {AverageFC( Value1, SlowLength ) ;}

SlowAvgMaTLAng = GZF_MATLAng(Price, NormRatio, SlowLength, TLLength);

Plot1( 90, "TLAngMAUp" ) ;
Plot2( 0, "TLAngMAFlat" ) ;
Plot3( -90, "TLAngMADn" ) ;
SetPlotColor( 1, ColorBase ) ;
SetPlotColor( 2, ColorBase ) ;
SetPlotColor( 3, ColorBase ) ;

Plot4( FastAvgTLAngMa, "TLAngMAFa" ) ;
SetPlotColor( 4, ColorTLAngMaFa ) ;

Plot5( SlowAvgTLAngMa, "TLAngMASl" ) ;
SetPlotColor( 5, ColorTLAngMaSl ) ;

Plot6( SlowAvgMaTLAng, "MATLAngSl" ) ;
SetPlotColor( 6, ColorMaTLAngSl ) ;
